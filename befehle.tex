% Clean chapter quotations
\newcommand{\hugequote}{%
{\book\fontsize{75}{80}\selectfont%
\hspace*{-.475em}\color{graylight}%
\textit{\glqq}%
\vskip -.26em}%
}
\newcommand{\chapterquote}[3]{%
\begin{minipage}{.865\textwidth}%
    \begin{flushright}
        \begin{minipage}{.65\textwidth}%
            \begin{flushleft}
                {\hugequote}\textit{#1}
            \end{flushleft}
            \begin{flushright}
                --- \textbf{#2} \\
                #3
            \end{flushright}
        \end{minipage}%
    \end{flushright}
\end{minipage}%
\bigskip
}
