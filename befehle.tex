    \newcommand{\ctchapternumber}[1]{%
        \usekomafont{chapter}%
        \begin{minipage}[t]{0.3\textwidth}%
            \raggedleft{%
                {\vspace*{-0.5cm}\color{ctcolorchapterline}\rule[-5pt]{2pt}{5.5cm}}%
                \quad%
                {\color{ctcolorchapternum}\fontsize{60}{60}\selectfont\textbf{#1}}%
            }%
        \end{minipage}%
    }

        \end{minipage}%
    }
